{% macro encode_decode(charlist=None, codes=None, mode="encode", scheme="ascii") %}
  {% set chunk_size = 10 %}
  {% set total = charlist|length %}
  {% if mode == "encode" %}
  \textbf{ {{tr("Encode the following text into")}} {{ scheme|upper }} }\\
  \textbf{Text:} \aRL{"{{ charlist|escape_string|join }}"}\\[6pt]
  {% elif mode == "decode" %}
      \textbf{  {{tr("Decode the following {scheme} codes into text", scheme=scheme|upper)}} }\\
      \textbf{ {{tr("Codes:")}} } {{ codes }}\\[6pt]
  {%endif%}
  {% for start in range(0, total, chunk_size) %}
    {% set end = start + chunk_size %}
    {% set sub_chars = charlist[start:end] %}
    {% set sub_codes = codes[start:end] %}

    {% if mode == "encode" %}
      \begin{tabular}{|c|{% for _ in sub_chars %}c|{% endfor %}}
        \hline
       {{tr("Character")}}
        {% for ch in sub_chars %}
          & {% if ch == " " %}
                space
            {% elif ch == "\n" %}
                newline
            {% elif ch == "\t" %}
                tab
            {% elif '\u0600' <= ch <= '\u06FF' %}
                \aRL{ {{ ch }} }
            {% else %}
                {{ ch|escape_string }}
            {% endif %}
        {% endfor %} \\
        \hline
        {{tr("{scheme} Code Point",scheme=scheme|upper)}}
        {% for cp in sub_codes %}
          & {{ cp }}
        {% endfor %} \\
        \hline
      \end{tabular}

    {% elif mode == "decode" %}


      \begin{tabular}{|c|{% for _ in sub_codes %}c|{% endfor %}}
        \hline
        {{ tr("{scheme} Code Point", scheme=scheme|upper)}}
        {% for cp in sub_codes %}
          & {{ cp }}
        {% endfor %} \\
        \hline
        {{tr("Character")}}
        {% for ch in sub_chars %}
          & {% if ch == " " %}
                space
            {% elif ch == "\n" %}
                newline
            {% elif ch == "\t" %}
                tab
            {% elif '\u0600' <= ch <= '\u06FF' %}
                \aRL{ {{ ch }} }
            {% else %}
                {{ ch|escape_string }}
            {% endif %}
        {% endfor %} \\
        \hline
      \end{tabular}
    {% endif %}

    \vspace{1em} % spacing between tables
  {% endfor %}
{% endmacro %}
